# 导航查询

### 一对一关系

```csharp
public class StudentA
{
    [SugarColumn(IsPrimaryKey = true, IsIdentity = true)]
    public int StudentId { get; set; }
    public string Name { get; set; }
    public int SchoolId { get; set; }
    
    // 用例 1：主键模式
    [Navigate(NavigateType.OneToOne, nameof(SchoolId))]
    public SchoolA SchoolA { get; set; }  // 不能赋值只能是 null
    
    // 用例 2：反向导航，两个字段匹配
    [Navigate(NavigateType.OneToOne, nameof(SchoolId), nameof(SchoolA.Id))]
    public SchoolA SchoolItem { get; set; }
    
    // 用例 3：字典导航
    [SqlSugar.Navigate(NavigateType.OneToOne, nameof(SexId), 
        nameof(DataDictionary1.Code), "type='sex'")]
    
    // 用例 4：默认值支持
    [Navigate(NavigateType.OneToOne, nameof(SchoolId))]
    public SchoolA SchoolA { get; set; } = new OneToOneInitializer<SchoolA>();
}

// 导航 + 主表过滤
var list = db.Queryable<StudentA>()
    .Includes(x => x.SchoolA)
    .Where(x => x.SchoolA.SchoolName == "北大")
    .ToList();

// 导航 + 只查有导航数据 (5.1.2.8)
var list = db.Queryable<StudentA>()
    .Includes(x => x.SchoolA)
    .Where(x => SqlFunc.Exists(x.SchoolA.主键))
    .ToList();

// 导航指定字段 (5.1.3.38)
var list = db.Queryable<StudentA>()
    .Includes(x => x.SchoolA.ToList(it => new SchoolA() { 
        Name = it.Name, 
        id = it.Id 
    }))
    .ToList();
```

### 一对多关系

```csharp
public class StudentA
{
    [SugarColumn(IsPrimaryKey = true, IsIdentity = true)]
    public int Id { get; set; }
    public string Name { get; set; }
    public int SchoolId { get; set; }
    
    // 用例 1：正常一对多
    [Navigate(NavigateType.OneToMany, nameof(BookA.studenId))]
    public List<BookA> Books { get; set; }
    
    // 用例 2：反向导航
    [Navigate(NavigateType.OneToMany, nameof(BookA.studenId), nameof(Id))]
    public List<BookA> Books { get; set; }
}

public class BookA
{
    [SugarColumn(IsPrimaryKey = true, IsIdentity = true)]
    public int BookId { get; set; }
    public string Name { get; set; }
    public int studenId { get; set; }
}

// 用法
// 例 1：简单用法
var list = db.Queryable<StudentA>()
    .Includes(x => x.Books)
    .ToList();

// 例 2：支持 Any 和 Count 对主表过滤
var list = db.Queryable<StudentA>()
    .Includes(x => x.Books)
    .Where(x => x.Books.Any())  // 或 .Where(x => x.Books.Any(z => z.Name == "jack"))
    .ToList();

// 例 3：Where 子对象排序和过滤
var list = db.Queryable<StudentA>()
    .Includes(x => x.Books
        .Where(y => y.BookId > 0)
        .OrderBy(y => y.BookId)
        .ToList())
    .ToList();

// 例 4：主表 + 子表都过滤
var list = db.Queryable<StudentA>()
    .Includes(x => x.Books.Where(it => it.Name == "jack").ToList())
    .Where(x => x.Books.Any(z => z.Name == "jack"))
    .ToList();

// 例 5：Select 指定字段
var list = db.Queryable<StudentA>()
    .Includes(x => x.Books
        .Select(z => new BookA() { Names = z.Names })
        .ToList())

// 例 6：Take 取前几条
var list = db.Queryable<StudentA>()
    .Includes(x => x.Books.Take(10).ToList())
    .ToList();
```

### 多对多关系

```csharp
public class ABMapping1
{
    [SugarColumn(IsPrimaryKey = true)]
    public int AId { get; set; }
    [SugarColumn(IsPrimaryKey = true)]
    public int BId { get; set; }
}

public class A1
{
    [SugarColumn(IsPrimaryKey = true, IsIdentity = true)]
    public int Id { get; set; }
    public string Name { get; set; }
    
    [Navigate(typeof(ABMapping1), nameof(ABMapping1.AId), nameof(ABMapping1.BId))]
    public List<B1> BList { get; set; }
}

public class B1
{
    [SugarColumn(IsPrimaryKey = true, IsIdentity = true)]
    public int Id { get; set; }
    public string Name { get; set; }
    
    [Navigate(typeof(ABMapping1), nameof(ABMapping1.BId), nameof(ABMapping1.AId))]
    public List<A1> AList { get; set; }
}

// 用法
var list3 = db.Queryable<A1>().Includes(x => x.BList).ToList();
```

### 多级导航

```csharp
var list2 = db.Queryable<StudentA>()
    .Includes(st => st.SchoolA, sch => sch.RoomList)  // 查询 2 级
    .Includes(st => st.Books)                         // 查询 1 级
    .ToList();

// 超过 3 个层级需要 AsNavQueryable
db.Queryable<Order>()
    .Includes(it => it.xx)
    .AsNavQueryable()
    .Includes(it => it.1, it => it.2, it => it.3, it => it.4, it => it.5)
```

### 转 Dto

```csharp
// 1. Mapster 工具映射（推荐）
var list = db.Queryable<StudentA>()
    .Includes(x => x.Books).ToList();
var dtoList = list.Adapt<List<StudentDto>>();

// 通过规则映射 Dto
public class TreeDto
{
    public int Id { get; set; }
    public string Name { get; set; }
    public int ParentId { get; set; }
    public string ParentName { get; set; }  // 对应 Parent 中的 Name
}

var list = db.Queryable<Tree>()
    .Includes(it => it.Parent)
    .ToList();
var dtolist = list.Adapt<List<TreeDto>>();

// 2. 手动转 Dto (5.1.4.71)
var list = db.Queryable<Student_004>()
    .Includes(x => x.books)
    .Select(x => new Student_004Dto {
        books = x.books
    }, true)  // true 是自动映射其他属性
    .ToList();

// Mapster 转换
var list = db.Queryable<Student_004>()
    .Includes(x => x.books)
    .Select(x => new Student_004Dto {
        books = x.books.Adapt<List<BooksDto>>()
    })
    .ToList();
```

### 导航方法

```csharp
// 一对多和多对多
// Count
Where(it => it.导航对象.Any())
Where(it => it.导航对象.Any(z => z.id == 1))
Where(it => it.导航对象.Any(List<IConditionalModel>))

// Any
Where(it => it.导航对象.Any())
Where(it => it.导航对象.Any(z => z.id == 1))

// 一对一 (5.1.2.9)
Where(x => SqlFunc.Exists(x.SchoolA.Id))
Where(x => SqlFunc.Exists(x.SchoolA.Id, List<IConditionalModel>))
```

### ThenMapper 用法

```csharp
// 普通导航无法实现时使用 ThenMapper
var list = db.Queryable<StudentA>().ToList();
db.ThenMapper(list, stu => {
    stu.SchoolA = db.Queryable<SchoolA>()
        .SetContext(scl => scl.SchoolId, () => stu.SchoolId, stu)
        .FirstOrDefault();
});

// SetContext 避免循环查询，高性能
```
