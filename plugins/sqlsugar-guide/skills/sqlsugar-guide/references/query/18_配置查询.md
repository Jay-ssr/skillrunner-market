# 配置查询

### 字典导航

```csharp
public class DataMain
{
    public int Id { get; set; }
    public string Name { get; set; }
    public string SexCode { get; set; }
    public string Province { get; set; }

    // 导航
    [SqlSugar.Navigate(NavigateType.OneToOne,
        nameof(SexCode), nameof(DataDictionary1.Code), "type='sex'")]
    public DataDictionary1 SexInfo { get; set; }

    [SqlSugar.Navigate(NavigateType.OneToOne,
        nameof(Province), nameof(DataDictionary1.Code), "type='province'")]
    public DataDictionary1 ProvinceInfo { get; set; }
}

var list = db.Queryable<DataMain>()
    .Includes(x => x.SexInfo)
    .Includes(x => x.ProvinceInfo)
    .ToList();

var list2 = db.Queryable<DataMain>()
    .Where(x => x.SexInfo.Name == "男")
    .ToList();
```

### 字典表配置

```csharp
// 字典实体
public class DataDictionary
{
    public string Code { get; set; }
    public string Name { get; set; }
    public string Type { get; set; }
}

// 配置字典表
if (!db.ConfigQuery.Any())  // 保证只配置一次
{
    var types = db.Queryable<DataDictionary>()
        .Select(it => it.Type)
        .Distinct()
        .ToList();

    foreach (var type in types)
    {
        db.ConfigQuery.SetTable<DataDictionary>(
            it => it.Code,      // 主键或关联字段
            it => it.Name,      // 显示的文本
            type,               // 唯一标识
            it => it.Type == type);  // 查询条件
    }
}

// 使用
var res = db.Queryable<Person>().Select(it => new Person() {
    Id = it.Id.SelectAll(),
    SexName = it.SexId.GetConfigValue<DataDictionary>("sex"),
    ProvinceName = it.ProvinceId.GetConfigValue<DataDictionary>("province"),
    CityName = it.CityId.GetConfigValue<DataDictionary>("city")
}).ToList();

// SQL
// SELECT *,
//   (SELECT name FROM DataDictionary WHERE type='sex' AND t.code=sexid) AS sexname,
//   (SELECT name FROM DataDictionary WHERE type='city' AND t.code=cityid) AS cityname,
//   (SELECT name FROM DataDictionary WHERE type='province' AND t.code=provinceid) AS provincename
// FROM Person t
```

### 简单联表配置

```csharp
// 配置 Order
db.ConfigQuery.SetTable<Order>(it => it.Id, it => it.Name);

// 使用
var list3 = db.Queryable<OrderItem>().Select(it => new OrderItem {
    ItemId = it.ItemId.SelectAll(),
    OrderName = it.OrderId.GetConfigValue<Order>()
}).ToList();
```
