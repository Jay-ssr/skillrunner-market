# 子查询

### 基础 API

```csharp
/*** 只查一列 ***/

// First: select top 1 id
SqlFunc.Subqueryable<School>()
    .Where(s => s.Id == st.Id)
    .Select(s => s.Id)

// Max
SqlFunc.Subqueryable<School>()
    .Where(s => s.Id == st.Id)
    .Max(s => s.Id)

// Min
SqlFunc.Subqueryable<School>()
    .Where(s => s.Id == st.Id)
    .Min(s => s.Id)

// Avg
SqlFunc.Subqueryable<School>()
    .Where(s => s.Id == st.Id)
    .Avg(s => s.Id)

// Count
SqlFunc.Subqueryable<School>()
    .Where(s => s.Id == st.Id)
    .Count()

// Sum
SqlFunc.Subqueryable<School>()
    .Where(s => s.Id == st.Id)
    .Sum(s => s.num)

// 去重后汇总数量
SqlFunc.Subqueryable<School>()
    .Where(s => z.Id == st.Id)
    .Select(s => SqlFunc.AggregateDistinctCount(s.Name))

// 逗号分割列
SqlFunc.Subqueryable<Order>()
    .Where(z => z.Id == it.Id)
    .SelectStringJoin(z => z.Name, ",")

/*** 返回 List 或者实体 ***/
// 新功能：5.1.3.36
SqlFunc.Subqueryable<Custom>()
    .Where(c => c.Id == it.CustomId)
    .ToList()

SqlFunc.Subqueryable<Custom>()
    .Where(c => c.Id == it.CustomId)
    .ToList(it => new Class() { Name = it.Name })

/*** 返回单个实体 ***/
SqlFunc.Subqueryable<Custom>()
    .Where(c => c.Id == it.CustomId)
    .First()
```

### Where 中使用子查询

```csharp
var getAll = db.Queryable<Student, School>((st, sc) => 
    new JoinQueryInfos(JoinType.Left, st.Id == sc.Id))
    .Where(st => st.Id == SqlFunc.Subqueryable<School>()
        .Where(s => s.Id == st.Id)
        .Select(s => s.Id))
    .ToList();

// SQL (MySQL)
SELECT `st`.`ID`, `st`.`SchoolId`, `st`.`Name`, `st`.`CreateTime`
FROM `STudent` st Left JOIN `School` sc ON (`st`.`ID` = `sc`.`Id`)
WHERE (`st`.`ID` = (SELECT `Id` FROM `School` WHERE (`Id` = `st`.`ID`) LIMIT 0,1))
```

### Select 中使用子查询

```csharp
var getAll = db.Queryable<Student, School>((st, sc) => 
    new JoinQueryInfos(JoinType.Left, st.Id == sc.Id))
    .Select(st => new {
        name = st.Name,
        id = SqlFunc.Subqueryable<School>()
            .Where(s => s.Id == st.Id)
            .Select(s => s.Id)
    }).ToList();

// 完整用例
Select(it => new {
    id = it.id,
    schoolname = SqlFunc.Subqueryable<School>()
        .Where(s => s.Id == it.ShoolId)
        .Select(s => s.Name)
    // SQL: schoolname = (SELECT `Name` FROM `School` WHERE (`s`.`Id` = `it`.`ShoolId`) LIMIT 0,1)
})
```

### IN 和 NOT IN

```csharp
// Exists（等同于 IN）
var getAll7 = db.Queryable<Student>()
    .Where(it => SqlFunc.Subqueryable<School>()
        .Where(s => s.Id == it.Id)
        .Any())
    .ToList();
// SQL: WHERE (EXISTS (SELECT * FROM `School` WHERE (`Id` = `it`.`ID`)))

// Not Exists（等同于 NOT IN）
var getAll8 = db.Queryable<Student>()
    .Where(it => SqlFunc.Subqueryable<School>()
        .Where(s => s.Id == it.Id)
        .NotAny())
    .ToList();

// In 和 Not In (5.0.7.5)
// 在有 GroupBy 的情况下自动转成 in
db.Queryable<Order>()
    .Where(it => it.Id == SqlFunc.Subqueryable<Custom>()
        .GroupBy(z => z.OrderId)
        .Select(z => z.OrderId))
    .ToList()

// 分页 In
db.Queryable<Order>()
    .In(it => it.Id, 
        db.Queryable<Order>().Skip(1).Take(100).Select(it => it.Id))
    .ToList();
```

### 子查询 ToList

```csharp
// 方式 1：简单方式 (5.1.3.38)
// 性能优于循环，但低于 ThenMapper
db.Queryable<Order>()
    .Select(it => new {
        CustomId = it.CustomId,
        OrderId = it.Id,
        OrderName = it.Name,
        CustomList = SqlFunc.Subqueryable<Custom>()
            .Where(c => c.Id == it.CustomId)
            .ToList()
    })
    .ToList();

// 也可以 Dto
SqlFunc.Subqueryable<Custom>()
    .Where(c => c.Id == it.CustomId)
    .ToList(c => new Class() { Name = c.Name })

// 只查 List<string>
SqlFunc.Subqueryable<Custom>()
    .Where(c => c.Id == it.CustomId)
    .ToList(c => c.Name)

// 自动 Dto (5.1.4.70)
SqlFunc.Subqueryable<Custom>()
    .Where(c => c.Id == it.CustomId)
    .ToList<Dto>()

// 方式 2：ThenMapper（推荐）
// 性能好，避免垃圾 SQL
var treeRoot = db.Queryable<Tree>().ToList();
db.ThenMapper(treeRoot, item => {
    item.Child = db.Queryable<Tree>()
        .SetContext(x => x.ParentId, () => item.Id, item)
        .ToList();
});

// 异步
var treeRoot = await db.Queryable<Tree>().ToListAsync();
await db.ThenMapperAsync(treeRoot, async item => {
    item.Child = await db.Queryable<Tree>()
        .SetContextAsync(x => x.ParentId, () => item.Id, item);
});

// 无限层级
// 第一层
db.ThenMapper(treeRoot, item => {
    item.Child = db.Queryable<Tree>()
        .SetContext(x => x.ParentId, () => item.Id, item)
        .ToList();
});
// 第二层
db.ThenMapper(treeRoot.SelectMany(it => it.Child), it => {
    it.Child = db.Queryable<Tree>()
        .SetContext(x => x.ParentId, () => it.Id, it)
        .ToList();
});
```

### 子查询在 Join

```csharp
// 5.0.5.4+
var list = db.Queryable<Order>()
    .Where(it => SqlFunc.Subqueryable<A>()
        .LeftJoin<B>((i, y) => i.ItemId == y.ItemId)
        .InnerJoin<C>((i, y, z) => i.ItemId == z.ItemId)
        .Where((i, y, z) => i.ItemId == it.Aid)
        .Any())
    .ToList();
```

### 子查询使用别名

```csharp
// 如果点不出 AS 请升级
SqlFunc.Subqueryable<A>().AS("A01")  // 查询 A01 表
```

### 字符串拼接

```csharp
// 5.1.1-preview14
// MySQL/SQLite: group_concat
// Oracle: listagg
// PgSQL: string_agg
var list = db.Queryable<Order>()
    .Select(it => new {
        names = SqlFunc.Subqueryable<custom>()
            .Where(z => z.oid == it.id)
            .SelectStringJoin(z => z.Name, ",")
    })
    .ToList();

// 多字段拼接
.SelectStringJoin(z => SqlFunc.MergeString(z.Name, "-", z.Id.ToString()), ",")

// 去重复
.SelectStringJoin(it => SqlFunc.MappingColumn<string>($"distinct {it.UserId}"), ".")
```
