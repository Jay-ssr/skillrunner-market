# 联表查询

### 语法糖一：新语法 Join

```csharp
// 联表查询
var query5 = db.Queryable<Order>()
    .LeftJoin<Custom>((o, cus) => o.CustomId == cus.Id)    // 多个条件用 &&
    .LeftJoin<OrderDetail>((o, cus, oritem) => o.Id == oritem.OrderId)
    .Where(o => o.Id == 1)
    .Select((o, cus, oritem) => new ViewOrder { 
        Id = o.Id, 
        CustomName = cus.Name 
    })
    .ToList();

// SQL
SELECT [o].[Id] AS [Id], [cus].[Name] AS [CustomName]
FROM [Order] o
Left JOIN [Custom] cus ON ([o].[CustomId] = [cus].[Id])
Left JOIN [OrderDetail] oritem ON ([o].[Id] = [oritem].[OrderId])
WHERE ([o].[Id] = @Id0)

// Join 类型
.LeftJoin       // 左连接
.InnerJoin      // 内连接
.FullJoin       // 全连接（需要高版本）

// 表和 Queryable JOIN
var rigtQueryable = db.Queryable<Custom>()
    .LeftJoin<OrderItem>((o, i) => o.Id == i.ItemId)
    .Select(o => o);

var list = db.Queryable<Order>()
    .LeftJoin(rigtQueryable, (c, j) => c.CustomId == j.Id)
    .Select(c => c).ToList();

// Queryable 和表 JOIN
var queryable = db.Queryable<Order>();
var list = db.Queryable(queryable)
    .LeftJoin<OrderDetails>((o, d) => o.id == d.orderid)
    .Select(o => o).ToList();
```

### 语法糖二：多类型参数

```csharp
// 适合联表比较多的场景
var list = db.Queryable<Order, OrderItem, Custom>(
    (o, i, c) => new JoinQueryInfos(
        JoinType.Left, o.Id == i.OrderId,
        JoinType.Left, o.CustomId == c.Id
    ))
    .Select((o, i, c) => new ViewModel { name = o.Name })
    .ToList();

// SQL
SELECT ... FROM [Order] o
Left JOIN [OrderItem] i ON ([o].[Id] = [i].[OrderId])
Left JOIN [Custom] c ON ([o].[CustomId] = [c].[Id])

// 支持 Cross Join
.JoinType.CrossJoin, it => true
```

### Where/OrderBy/GroupBy 用法

```csharp
// 注意：写在 Select 之前

// Where
.Where(o => o.id == 1)        // 只用到 o
.Where((o, i) => i.xx == 1)  // 如果用到 i

// OrderBy
.OrderBy(o => o.id)           // 只用到 o
.OrderBy((o, i) => i.xx)     // 如果用到 i

// GroupBy
.GroupBy(o => o.id)           // 只用到 o
.GroupBy((o, i) => i.xx)     // 如果用到 i
```

### 导航属性联表

```csharp
// 配置导航属性后可省略 Join
public class StudentA
{
    [SugarColumn(IsPrimaryKey = true, IsIdentity = true)]
    public int StudentId { get; set; }
    public string Name { get; set; }
    public int SchoolId { get; set; }
    
    [Navigate(NavigateType.OneToOne, nameof(SchoolId))]
    public SchoolA SchoolA { get; set; }  // 不能赋值只能是 null
}

// IncludeLeftJoin 会根据导航生成 LeftJoin 语句
var list = db.Queryable<StudentA>()
    .IncludeLeftJoin(x => x.SchoolA)  // Left Join SchoolA y ON x.SchoolId = y.Id
    .Where(x => x.SchoolA.SchoolName == "北大")
    .ToList();

// 没有 IncludeLeftJoin 也可以在 Where 中使用（SQL 是子查询过滤）
var list = db.Queryable<StudentA>()
    .Where(x => x.SchoolA.SchoolName == "北大")
    .ToList();

// 没有 IncludeLeftJoin 也可以在 Select 中使用（SQL 是子查询实现）
var list = db.Queryable<StudentA>()
    .Where(x => x.id > 1)
    .Select(x => new {
        name = x.Name,
        SchoolName = x.SchoolA.SchoolName
    }).ToList();
```

### 设置别名

```csharp
var list1 = db.Queryable<Order>()
    .AS("Order001")
    .LeftJoin<OrderItem>((o, i) => o.Id == i.OrderId, "OrderItem001")
    .LeftJoin<Custom>((o, i, c) => c.Id == o.CustomId, "Custom001")
```

### 超过 12 个表的联表

```csharp
// 通过 MergeTable 合并成一个表，然后再进行 JOIN
db.Queryable<Order>()
    .LeftJoin<OrderItem>((x, y) => x.id == y.ItemId)
    .LeftJoin.....  // 更多 JOIN
    .Select((x, y, z, ...) => new { xid = x.id, yid = y.ItemId })
    .MergeTable()   // 合并
    .LeftJoin<OrderItem>((x, y) => x.yid == y.ItemId)  // 最后一个表不能是匿名对象
    .ToList();
```

### 动态 Join

```csharp
// 条件动态
var exp2 = Expressionable.Create<Order, Custom>();
exp2.And((o, cus) => o.CustomId == cus.Id);
exp2.AndIF(Name != null, (o, cus) => o.Name == Name);

var list = db.Queryable<Order>()
    .LeftJoin<Custom>(exp2.ToExpression())  // 动态拼出来的条件
    .LeftJoin<OrderDetail>((o, cus, oritem) => o.Id == oritem.OrderId)
    .Where(o => o.Id == 1)
    .ToList();

// 表动态
var list = db.Queryable<Order>()
    .LeftJoinIF<Custom>(条件, (o, cus) => o.CustomId == cus.Id)  // 条件成立才生成 JOIN
    .LeftJoin<OrderDetail>((o, cus, oritem) => o.Id == oritem.OrderId)
    .ToList();
```
