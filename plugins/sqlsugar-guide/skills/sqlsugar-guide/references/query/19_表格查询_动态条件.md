# 表格查询（动态条件）

### 简单后台构造

```csharp
var conModels = new List<IConditionalModel>();
conModels.Add(new ConditionalModel {
    FieldName = "id",
    ConditionalType = ConditionalType.Equal,
    FieldValue = "1"
});
var student = db.Queryable<Student>().Where(conModels).ToList();

// 指定类型
IConditionalModel conditionalModel = new ConditionalModel() {
    ConditionalType = 0,
    FieldName = "id",
    FieldValue = "1",
    CSharpTypeName = "int"  // 设置类型
};
```

### JSON 构造

```csharp
// JSON 格式
[
    {"FieldName": "id", "ConditionalType": "0", "FieldValue": "1"},
    {"FieldName": "name", "ConditionalType": "0", "FieldValue": "jack"}
]

// 带类型
[
    {"FieldName": "id", "ConditionalType": "0", "FieldValue": "1", "CSharpTypeName": "int"}
]

// 5.0.5.1 Json 直接序列化
var conModels = db.Utilities.JsonToConditionalModels(json);
var student = db.Queryable<Student>().Where(conModels).ToList();
// SQL: SELECT * FROM Student WHERE id=1 AND name='jack'
```

### 二级构造

```csharp
// 后台构造
var conModels = new List<IConditionalModel>();
conModels.Add(new ConditionalModel {
    FieldName = "name",
    ConditionalType = ConditionalType.Equal,
    FieldValue = "jack"
});
conModels.Add(new ConditionalModels() {
    ConditionalList = new List<KeyValuePair<WhereType, ConditionalModel>>() {
        new KeyValuePair<WhereType, ConditionalModel>(
            WhereType.And,
            new ConditionalModel() { FieldName = "id", ConditionalType = ConditionalType.Equal, FieldValue = "1" }),
        new KeyValuePair<WhereType, ConditionalModel>(
            WhereType.Or,
            new ConditionalModel() { FieldName = "id", ConditionalType = ConditionalType.Equal, FieldValue = "2" }),
        new KeyValuePair<WhereType, ConditionalModel>(
            WhereType.And,
            new ConditionalModel() { FieldName = "id", ConditionalType = ConditionalType.Equal, FieldValue = "3" })
    }
});
// SQL: name='jack' AND (id=1 OR id=2 AND id=3)

// 前端构造
[
    {"FieldName": "id", "FieldValue": "1", "ConditionalType": 10},
    {"FieldName": "id", "FieldValue": null, "ConditionalType": 12},
    {
        "ConditionalList": [{
            "Key": 1,  // 1=Or, 0=And
            "Value": {"FieldName": "id", "FieldValue": "2", "ConditionalType": 0}
        }]
    }
]
```

### 树型构造

```csharp
// 支持任意层级条件
[{
    "ConditionalList": [{
        "Key": -1,  // null = -1
        "Value": {"FieldName": "id", "FieldValue": "2", "ConditionalType": 0}
    }, {
        "Key": 0,  // And = 0
        "Value": {"FieldName": "name", "FieldValue": "2", "ConditionalType": 0}
    }, {
        "Key": 0,
        "Value": {
            "ConditionalList": [{
                "Key": -1,
                "Value": {"FieldName": "price", "FieldValue": "1", "ConditionalType": 0}
            }, {
                "Key": 0,
                "Value": {"FieldName": "CustomId", "FieldValue": "1", "ConditionalType": 0}
            }]
        }
    }]
}]

var conditionalModels = db.Utilities.JsonToConditionalModels(json);
var list = db.Queryable<Order>().Where(conditionalModels).ToList();
```

### 操作符说明

| ConditionalType | 值 | 描述 |
|-----------------|-----|------|
| Equal | 0 | 等于 |
| Like | 1 | 模糊查询 |
| GreaterThan | 2 | 大于 |
| GreaterThanOrEqual | 3 | 大于等于 |
| LessThan | 4 | 小于 |
| LessThanOrEqual | 5 | 小于等于 |
| In | 6 | IN，格式：X,Y,Z |
| NotIn | 7 | NOT IN |
| LikeLeft | 8 | 左模糊 |
| LikeRight | 9 | 右模糊 |
| NoEqual | 10 | 不等于 |
| IsNullOrEmpty | 11 | 是 null 或 '' |
| IsNot | 12 | 不等于 null 或 is not null |
| NoLike | 13 | 模糊查询取反 |
| EqualNull | 14 | 等于 null 或等于值 |
| InLike | 15 | IN LIKE |
| Range | 16 | 范围，格式：1,2 |
| DateRange | 17 | 日期范围 |

### 类型转换

```csharp
// PGQL 不支持字符串与 INT 相等
IConditionalModel conditionalModel = new ConditionalModel() {
    ConditionalType = 0,
    FieldName = "id",
    FieldValue = "1",
    CSharpTypeName = "int"
};
```

### 多表查询

```csharp
List<IConditionalModel> conModels = new List<IConditionalModel>();
conModels.Add(new ConditionalModel { 
    FieldName = "id", 
    ConditionalType = ConditionalType.Equal,
    FieldValue = "1"
});

var oneClass = db.Queryable<Order, OrderItem, Custom>((o, i, c) => new JoinQueryInfos(
    JoinType.Left, o.Id == i.OrderId,
    JoinType.Left, o.CustomId == c.Id
))
.Select((o, i, c) => new ViewOrder { Id = o.Id, CustomName = c.Name })
.MergeTable()  // 通过 MergeTable 变成新表
.Where(conModels)
.ToList();
```

### 导航+动态查询

```csharp
// 一对一
.Where(List<IConditionalModel>)  // 主表过滤
.Where(x => SqlFunc.Exists(x.SchoolA.Id, List<IConditionalModel>))  // 导航表过滤

// 多对多或一对多
.Where(List<IConditionalModel>)  // 主表过滤
.Where(it => it.导航对象.Any(List<IConditionalModel>))  // 导航表过滤
```
