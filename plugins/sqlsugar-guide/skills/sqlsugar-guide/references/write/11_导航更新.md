# 导航更新

### 导航更新语法（多层级）

```csharp
List<Student> list = new List<Student>();
list.Add(new Student() {
    Id = 1,
    Name = "jack",
    SchoolId = 2,
    SchoolA = new SchoolA() { ... RoomList = new List<Room> { ... } },
    Books = new List<Books> { ... }
});

db.UpdateNav(list)
    .Include(z1 => z1.SchoolA).ThenInclude(z1 => z1.RoomList) //更新2层
    .Include(z1 => z1.Books) //更新第1层
    .ExecuteCommand();
```

### 一对一

#### 逻辑

更新主表；子表已存在则更新，不存在则插入。

```csharp
//实体
public class StudentA
{
    [SugarColumn(IsPrimaryKey = true, IsIdentity = true)]
    public int StudentId { get; set; }
    public string Name { get; set; }
    public int SchoolId { get; set; }

    [Navigate(NavigateType.OneToOne, nameof(SchoolId))]
    public SchoolA SchoolA { get; set; }
}

var list = new List<StudentA>() { ... };
db.UpdateNav(list)
    .Include(z1 => z1.SchoolA)
    .ExecuteCommand();
```

#### 脏数据兼容（5.1.4.111-preview15+）

```csharp
var StudentA = new StudentA() {
    Id = 1,
    Name = "张三",
    SchoolId = 4, //如果这个ID=4数据库中不存在，只要有这个可能性，全部强制改成默认值0
    SchoolA = new SchoolA() {
        Id = 1, //这边赋值正确值就行，存在更新不存在插入
        Name = "北大"
    }

```

### 一对多

#### 模式1：子表先删后加

```csharp
//性能好，推荐使用
db.UpdateNav(list)
    .Include(x => x.Persons) //默认
    .ExecuteCommand();
```

#### 模式2：子表插入、更新或删除（5.1.4.113-preview+）

```csharp
//减少删除操作，但存在插入、更新、删除三种逻辑，性能略低于模式1
db.UpdateNav(list2)
    .Include(x => x.Persons, new SqlSugar.UpdateNavOptions() {
        OneToManyInsertOrUpdate = true, //配置启用插入、更新或删除模式
    })
    .ExecuteCommand();
```

#### 详细用例

```csharp
public class StudentA
{
    [SugarColumn(IsPrimaryKey = true, IsIdentity = true)]
    public int Id { get; set; }
    public string Name { get; set; }
    public int SchoolId { get; set; }

    [Navigate(NavigateType.OneToMany, nameof(BookA.studenId))]
    public List<BookA> Books { get; set; }
}

public class BookA
{
    [SugarColumn(IsPrimaryKey = true, IsIdentity = true)]
    public int BookId { get; set; }
    public string Name { get; set; }
    public int studenId { get; set; }
}

//先更新主表然后删除子表在插入子表
var list = new List<StudentA>() { ... };
db.UpdateNav(list)
    .Include(z1 => z1.Books)
    .ExecuteCommand();

//强制删除下级一对多脏数据（5.0.2.3-preview04）
db.UpdateNav(list)
    .Include(z1 => z1.Books, new UpdateNavOptions() { OneToManyDeleteAll = true })
    .ThenInclude(z1 => z1.RoomList)
    .ExecuteCommand();

//只更新子表
db.UpdateNav(list, new UpdateNavRootOptions() { IsDisableUpdateRoot = true })
    .Include(x => x.datas)
    .ExecuteCommand();

//假删除(5.1.4.86)
db.QueryFilter.AddTableFilter<Province>(it => it.IsDeleted == false);
db.UpdateNav(country)
    .Include(it => it.Provinces, new UpdateNavOptions() { OneToManyEnableLogicDelete = true })
    .ExecuteCommand();

//null列不更新5.1.144-preview13+
db.UpdateNav(list)
    .Include(x => x.Books, new SqlSugar.UpdateNavOptions() {
        OneToManyInsertOrUpdate = true,
        IgnoreNullColumns = true
    })
    .ExecuteCommand();

//忽略列不更新
db.UpdateNav(list)
    .Include(x => x.Books, new SqlSugar.UpdateNavOptions() {
        OneToManyInsertOrUpdate = true,
        IgnoreColumns = new string[] { "Name" }
    })
    .ExecuteCommand();
```

### 多对多

#### 逻辑

| 模式 | 说明 |
|------|------|
| 只更新关系表 | 默认模式（删除添加） |
| 更新A表+关系表 | `ManyToManyIsUpdateA = true` |
| 更新B表+关系表 | `ManyToManyIsUpdateB = true` |
| 更新A表+B表+关系表 | 同时设置 A 和 B |

#### 用例

```csharp
//默认模式：只更新关系表（删除添加）
db.UpdateNav(list)
    .Include(z1 => z1.BList)
    .ExecuteCommand();

//更新A表+更新关系表
db.UpdateNav(list)
    .Include(z1 => z1.BList, new UpdateNavOptions() {
        ManyToManyIsUpdateA = true
    })
    .ExecuteCommand();

//不存在添加B表,存在更新B表+更新关系
db.UpdateNav(list)
    .Include(z1 => z1.BList, new UpdateNavOptions() {
        ManyToManyIsUpdateB = true
    })
    .ExecuteCommand();

//更新A表+不存在添加B表,存在更新B表+更新关系
db.UpdateNav(list)
    .Include(z1 => z1.BList, new UpdateNavOptions() {
        ManyToManyIsUpdateA = true,
        ManyToManyIsUpdateB = true
    })
    .ExecuteCommand();
```

#### 中间额外字段赋值（5.1.4.86）

```csharp
db.UpdateNav(Opt)
    .Include(z => z.Roles, new UpdateNavOptions() {
        ManyToManySaveMappingTemplate = new OptRole() {
            CreateTime = "1010",
            OrgId = "1x"
        }
    })

```

#### 多对多假删除（5.1.4.89-preview10）

```csharp
db.UpdateNav(list)
    .Include(z1 => z1.BList, new UpdateNavOptions() {
        ManyToManyEnableLogicDelete = true //需要结合过滤器实现假删
    })
    .ExecuteCommand();
```

### 主表通用功能

```csharp
public class UpdateNavRootOptions
{
    public string[] IgnoreColumns { get; set; } //主表更新忽略列
    public string[] UpdateColumns { get; set; } //只更新哪几列
    public bool IsInsertRoot { get; set; } //强制插入主表，可以实现更新或者插入效果
    public string[] IgnoreInsertColumns { get; set; } //主表启用插入忽略列
    public bool IsIgnoreAllNullColumns { get; set; } //不更新null列
    public bool IsDiffLogEvent { get; set; } //启用主表差异日志
    public object DiffLogBizData { get; set; } //差异日志参数
    public bool IsDisableUpdateRoot { get; set; } //禁更新主表
    public bool IsOptLock { get; set; } //主表乐观锁
}
```

#### 忽略更新列

```csharp
db.UpdateNav(list, new UpdateNavRootOptions() {
    IgnoreColumns = new string[] { "Name", "CreateTime" }
})
    .Include(x => x.Roles)
    .ExecuteCommand();
```

#### 只更新列

```csharp
db.UpdateNav(list, new UpdateNavRootOptions() {
    UpdateColumns = new string[] { "Name", "CreateTime" }
})
    .Include(x => x.Roles)
    .ExecuteCommand();
```

#### 可以插入主表

```csharp
db.UpdateNav(list, new UpdateNavRootOptions() {
    IsInsertRoot = true
})
    .Include(x => x.Roles)
    .ExecuteCommand();
```

#### 禁止更新主表

```csharp
db.UpdateNav(list, new UpdateNavRootOptions() {
    IsDisableUpdateRoot = true
})
    .Include(x => x.Roles)
    .ExecuteCommand();
```

#### 乐观锁

```csharp
db.UpdateNav(list, new UpdateNavRootOptions() {
    IsOptLock = true
})
    .Include(x => x.Roles)
    .ExecuteCommand();
```

### 常见问题

多个共用一个对象的情况，可以用 `TranslateCopy` 处理：

```csharp
var tasks = db.Queryable<RatingTask>().Where(t => t.Id == 1 || t.Id == 2).ToList();
List<RatingTaskEmp> emps = new List<RatingTaskEmp> {
    new RatingTaskEmp { EmpId = 101, EmpName = "张三" },
    new RatingTaskEmp { EmpId = 102, EmpName = "李四" }
};
foreach (var task in tasks) {
    task.EmpList = emps; //多个共用一个对象
}

//需要将Task.EmpList都变成新对象
var tasks = db.Utilities.TranslateCopy(tasks);
```
